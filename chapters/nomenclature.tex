% ===================== Nomenclature entries =====================
\nomenclature{$A$}{Pre-exponential (frequency) factor; also area [m$^2$]}
\nomenclature{$c_p,\ c_v$}{Specific heat at constant pressure / volume [J\,kg$^{-1}$\,K$^{-1}$]}
\nomenclature{$c$}{Reaction progress variable (premixed), $0\le c\le1$}
\nomenclature{$D_k$}{Mass diffusion coefficient of species $k$ [m$^2$\,s$^{-1}$]}
\nomenclature{$E_a$}{Activation energy [J\,mol$^{-1}$]}
\nomenclature{$G$}{Gibbs free energy [J]}
\nomenclature{$h$}{Specific enthalpy [J\,kg$^{-1}$]}
\nomenclature{$\Delta h_{f}^{0}$}{Standard enthalpy of formation [J\,kg$^{-1}$ or J\,mol$^{-1}$]}
\nomenclature{$\Delta H_R$}{Enthalpy (heat) of reaction [J\,mol$^{-1}$]}
\nomenclature{$k$}{Reaction rate coefficient; turbulent kinetic energy [m$^2$\,s$^{-2}$]}
\nomenclature{$K_p,\ K_c$}{Equilibrium constant (pressure / concentration based)}
\nomenclature{$\ell_t$}{Integral length scale of turbulence [m]}
\nomenclature{LHV}{Lower heating value [J\,kg$^{-1}$]}
\nomenclature{$\dot m$}{Mass flow rate [kg\,s$^{-1}$]}
\nomenclature{$p$}{Pressure [Pa]}
\nomenclature{$R_0$}{Universal gas constant, $8.314$ J\,mol$^{-1}$\,K$^{-1}$}
\nomenclature{$s_L$}{Laminar burning (flame) velocity [m\,s$^{-1}$]}
\nomenclature{$s_T$}{Turbulent burning velocity [m\,s$^{-1}$]}
\nomenclature{$T$}{Temperature [K]}
\nomenclature{$T_{\mathrm{ad}}$}{Adiabatic flame temperature [K]}
\nomenclature{$u'$}{Turbulence intensity (r.m.s.\ velocity) [m\,s$^{-1}$]}
\nomenclature{$W,\ W_k$}{Mean / species $k$ molar mass [kg\,mol$^{-1}$]}
\nomenclature{$X_k$}{Mole fraction of species $k$}
\nomenclature{$Y_k$}{Mass fraction of species $k$}
\nomenclature{$Z$}{Mixture fraction; $Z_{\mathrm{st}}$ stoichiometric value}
\nomenclature{$\alpha$}{Thermal diffusivity [m$^2$\,s$^{-1}$]}
\nomenclature{$\chi$}{Scalar dissipation rate [s$^{-1}$]}
\nomenclature{$\delta_L$}{Laminar flame thickness [m]}
\nomenclature{$\varepsilon$}{Turbulent kinetic energy dissipation rate [m$^2$\,s$^{-3}$]}
\nomenclature{$\eta_k$}{Kolmogorov length scale [m]}
\nomenclature{$\kappa$}{Flame stretch rate [s$^{-1}$]}
\nomenclature{$\lambda$}{Thermal conductivity [W\,m$^{-1}$\,K$^{-1}$]}
\nomenclature{$\mu$}{Dynamic viscosity [Pa\,s]}
\nomenclature{$\nu_k',\ \nu_k''$}{Stoichiometric coefficients (reactant / product)}
\nomenclature{$\rho$}{Density [kg\,m$^{-3}$]}
\nomenclature{$\tau_c,\ \tau_t$}{Chemical / turbulent (integral) time scale [s]}
\nomenclature{$\phi$}{Equivalence ratio}
\nomenclature{$\dot\omega_k$}{Net chemical production rate of species $k$ [kg\,m$^{-3}$\,s$^{-1}$]}
\nomenclature{Da}{Damk\"ohler number, $\tau_t/\tau_c$}
\nomenclature{Ka}{Karlovitz number, $\tau_c/\tau_\eta$}
\nomenclature{Le}{Lewis number, $\alpha/D$}
\nomenclature{Pr}{Prandtl number, $\nu/\alpha$}
\nomenclature{Re}{Reynolds number}
\nomenclature{Sc}{Schmidt number, $\nu/D$}
